// ============================================================ // KY TaskManager - Google Apps Script Web App // スプレッドシートID（デプロイ前にここを書き換える） // ============================================================ const SHEET_TASKS = 'tasks'; const SHEET_REQUESTS = 'requests'; const SHEET_ORDERS = 'orders'; // ---------- CORS対応レスポンス ---------- function response(data) { // MimeType.TEXTにするとCORSプリフライトなしで通る return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.TEXT); } // ---------- シート初期化（初回実行で呼ぶ） ---------- function setupSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // tasks sheet let ts = ss.getSheetByName(SHEET_TASKS) || ss.insertSheet(SHEET_TASKS); if (ts.getLastRow() === 0) { ts.appendRow(['id','section','text','quadrant','status','created_at','done_at']); // 初期データ const now = new Date().toISOString().slice(0,10); const init = [ [1,'K','運転免許更新','q1','active',now,''], [2,'K','ファイル完成','q1','active',now,''], [3,'K','しょっぴー登録','q1','active',now,''], [4,'K','町内のデータ','q1','active',now,''], [5,'K','送料設定のやつ','q1','active',now,''], [6,'K','あさみの書類','q1','active',now,''], [7,'K','定期','q1','active',now,''], [8,'K','ラファ病院','q2','active',now,''], [9,'K','新川さんに出す','q2','active',now,''], ]; init.forEach(r => ts.appendRow(r)); } // requests sheet let rs = ss.getSheetByName(SHEET_REQUESTS) || ss.insertSheet(SHEET_REQUESTS); if (rs.getLastRow() === 0) { rs.appendRow(['id','from_user','to_user','text','status','created_at','done_at']); } // orders sheet let os = ss.getSheetByName(SHEET_ORDERS) || ss.insertSheet(SHEET_ORDERS); if (os.getLastRow() === 0) { os.appendRow(['order_id','item_name','buyer','amount_usd','step','assignee','created_at','notes']); } return 'Setup complete!'; } // ---------- GET ---------- function doGet(e) { try { const action = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'getTasks') { const data = sheetToObjects(ss.getSheetByName(SHEET_TASKS)); return response({ ok: true, data }); } if (action === 'getRequests') { const data = sheetToObjects(ss.getSheetByName(SHEET_REQUESTS)); return response({ ok: true, data }); } if (action === 'getOrders') { const data = sheetToObjects(ss.getSheetByName(SHEET_ORDERS)); return response({ ok: true, data }); } if (action === 'sendChatwork') { const token = body.token; const room = body.room; const msg = body.message; if (!token || !room || !msg) return response({ ok: false, error: 'Missing params' }); const url = `https://api.chatwork.com/v2/rooms/${room}/messages`; const res = UrlFetchApp.fetch(url, { method: 'post', headers: { 'X-ChatWorkToken': token }, payload: { body: msg }, muteHttpExceptions: true }); return response({ ok: true, status: res.getResponseCode() }); } return response({ ok: false, error: 'Unknown action' }); } catch(err) { return response({ ok: false, error: err.message }); } } // ---------- POST ---------- function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); // ---- タスク ---- if (action === 'addTask') { const sheet = ss.getSheetByName(SHEET_TASKS); const id = Date.now(); const now = new Date().toISOString().slice(0,10); sheet.appendRow([id, body.section, body.text, body.quadrant, 'active', now, '']); return response({ ok: true, id }); } if (action === 'updateTask') { const sheet = ss.getSheetByName(SHEET_TASKS); const row = findRow(sheet, String(body.id)); if (!row) return response({ ok: false, error: 'Row not found' }); if (body.status) sheet.getRange(row, 5).setValue(body.status); if (body.quadrant) sheet.getRange(row, 4).setValue(body.quadrant); if (body.text) sheet.getRange(row, 3).setValue(body.text); if (body.status === 'done') { sheet.getRange(row, 7).setValue(new Date().toISOString().slice(0,10)); } return response({ ok: true }); } if (action === 'doneTaskByText') { // テキスト一致で全セクションのタスクを完了 const sheet = ss.getSheetByName(SHEET_TASKS); const data = sheet.getDataRange().getValues(); const now = new Date().toISOString().slice(0,10); let count = 0; for (let i = 1; i < data.length; i++) { if (String(data[i][2]).trim() === String(body.text).trim() && data[i][4] === 'active') { sheet.getRange(i+1, 5).setValue('done'); sheet.getRange(i+1, 7).setValue(now); count++; } } return response({ ok: true, count }); } if (action === 'deleteTask') { const sheet = ss.getSheetByName(SHEET_TASKS); const row = findRow(sheet, String(body.id)); if (row) sheet.deleteRow(row); return response({ ok: true }); } // ---- 依頼 ---- if (action === 'addRequest') { const sheet = ss.getSheetByName(SHEET_REQUESTS); const id = Date.now(); const now = new Date().toISOString().slice(0,10); sheet.appendRow([id, body.from_user, body.to_user, body.text, 'pending', now, '']); return response({ ok: true, id }); } if (action === 'updateRequest') { const sheet = ss.getSheetByName(SHEET_REQUESTS); const row = findRow(sheet, String(body.id)); if (!row) return response({ ok: false, error: 'Row not found' }); if (body.status) { sheet.getRange(row, 5).setValue(body.status); if (body.status === 'done') { sheet.getRange(row, 7).setValue(new Date().toISOString().slice(0,10)); } } return response({ ok: true }); } if (action === 'deleteRequest') { const sheet = ss.getSheetByName(SHEET_REQUESTS); const row = findRow(sheet, String(body.id)); if (row) sheet.deleteRow(row); return response({ ok: true }); } // ---- 発送工程 ---- if (action === 'addOrder') { const sheet = ss.getSheetByName(SHEET_ORDERS); const now = new Date().toISOString().slice(0,10); sheet.appendRow([ body.order_id, body.item_name, body.buyer, body.amount_usd || '', 1, body.assignee || '', now, body.notes || '' ]); return response({ ok: true }); } if (action === 'updateOrderStep') { const sheet = ss.getSheetByName(SHEET_ORDERS); const row = findRow(sheet, body.order_id, 0); // col0=order_id if (!row) return response({ ok: false, error: 'Order not found' }); sheet.getRange(row, 5).setValue(body.step); if (body.notes) sheet.getRange(row, 8).setValue(body.notes); return response({ ok: true }); } if (action === 'deleteOrder') { const sheet = ss.getSheetByName(SHEET_ORDERS); const row = findRow(sheet, body.order_id, 0); if (row) sheet.deleteRow(row); return response({ ok: true }); } if (action === 'sendChatwork') { const token = body.token; const room = body.room; const msg = body.message; if (!token || !room || !msg) return response({ ok: false, error: 'Missing params' }); const url = `https://api.chatwork.com/v2/rooms/${room}/messages`; const res = UrlFetchApp.fetch(url, { method: 'post', headers: { 'X-ChatWorkToken': token }, payload: { body: msg }, muteHttpExceptions: true }); return response({ ok: true, status: res.getResponseCode() }); } return response({ ok: false, error: 'Unknown action' }); } catch(err) { return response({ ok: false, error: err.message }); } } // ---------- ユーティリティ ---------- function sheetToObjects(sheet) { if (!sheet) return []; const values = sheet.getDataRange().getValues(); if (values.length <= 1) return []; const headers = values[0]; return values.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function findRow(sheet, id, colIndex = 0) { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][colIndex]) === String(id)) return i + 1; } return null; } // ---------- eBay自動取込（将来用） ---------- // eBayのCSVをDropboxやGoogleドライブに置いて、 // このトリガーで定期チェックする想定 function importFromEbayCSV() { // TODO: eBay受注CSVをDriveから読んでordersシートに追加 // 現在は手動addOrderで運用 }
